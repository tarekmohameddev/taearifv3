export function useDragListener(
  type: EventKeys,
  fn: Events[EventKeys],
  deps: any[] = [],
) {
  const { setDragListeners } = useContext(dragListenerContext);

  useEffect(() => {
    if (setDragListeners) {
      setDragListeners((old) => ({
        ...old,
        [type]: [...(old[type] || []), fn],
      }));
    }
  }, deps);
}
